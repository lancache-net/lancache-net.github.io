Get up and running in seconds.

Quick-start Instructions

~ $ export LANCACHE_IP=`hostname -I | cut -d' ' -f1`

~ $ mkdir -p /cache/data /cache/logs

~ $ cd /cache

/cache $ docker run --restart unless-stopped --name lancache -v /cache/data:/data/cache -v /cache/logs:/data/logs -p $LANCACHE_IP:80:80 lancachenet/monolithic:latest

/cache $ docker run --restart unless-stopped --name lancache-dns -p $LANCACHE_IP:53:53/udp -e USE_GENERIC_CACHE=true -e LANCACHE_IP=$LANCACHE_IP lancachenet/lancache-dns:latest

/cache $ docker run --restart unless-stopped --name lancache-sniproxy -p $LANCACHE_IP:443:443 lancachenet/sniproxy:latest

/cache $ echo Please configure your dhcp server to serve dns as $HOST_IP

# => Check the correct IP is displayed above and configure your router for DNS
